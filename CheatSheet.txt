git init ( inicia un repositorio)

git status (muestra el estado de los archivos en su repositorio)

git add nombre-archivo (agrega un archivo y los pone seguimiento)
git add . (agrega todos los archivos y los pone en seguimiento)

git commit -m "descripcion del commit" (se usa para momitear un archivo)

git log (muestra historial de commits)

git pull (actualizar archivos contra la rama local)

git clone "url del repositorio" para clonar un repositorio existente

git remote

git remote -v 

mkdir (crea un directorio)

cd (acceder a una carpeta)

git branch (muestra las ramas)

git branch nombre-nueva-rama (crea una nueva rama)

git pull origin nombre-rama

git checkout nombre-rama (accede a la rama del nombre)

mkdir nombre_carpeta

cd (entra a una carpeta)

cd .. (salir carpeta)

ls 

clear (limpiar consola )
